39/**

42int check_philo_death(t_philo *philo)
43{
44 long current_time;
45 long time_since_meal;
46
47 pthread_mutex_lock(&philo->table->meal_lock);
48 current_time = get_time();
49 time_since_meal = current_time - philo->last_meal_time;
50 pthread_mutex_unlock(&philo->table->meal_lock);
51 if (time_since_meal >= philo->table->time_to_die)
52 {
53 print_status(philo, MSG_DIED);
