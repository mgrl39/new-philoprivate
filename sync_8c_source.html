44/**
45 * We are going to lock the table mutex
46 * and a pointer to the actual value
47 *
48 * increase threads running to synchro
49 * with the monitor.
50 *
51 * The monitor will use all_threads_running to
52 * indeed fetch this value and make a comparison
53 * if the nmber of threads running is equal to
54 * the actual total number of threads.
55 */
56int increase_long(t_mtx *mutex, long *value)
57{
58 if (pthread_mutex_lock(mutex))
59 return (ft_alert(FAIL_LOCK_INC_LONG, A_ERROR));
60 (*value)++;
61 if (pthread_mutex_unlock(mutex))
62 return (ft_alert(FAIL_UNLOCK_INC_LONG, A_ERROR));
63 return (SUCCESS);
64}
