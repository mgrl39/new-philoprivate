18void print_status(t_philo *philo, char *msg)
19{
20 long timestamp;
21
22 if (!philo || !msg || !philo->table)
23 return ;
24 pthread_mutex_lock(&philo->table->write_lock);
25 if (!check_death_flag(philo->table))
26 {
27 timestamp = get_time();
28 if (timestamp != -1)
29 {
30 timestamp -= philo->table->start_time;
31 printf("%06ld %d ", timestamp, philo->id);
32 if (ft_strcmp(msg, MSG_FORK) == 0)
33 printf(GOLD "%s" RESET "\n", msg);
34 else if (ft_strcmp(msg, MSG_EAT) == 0)
35 printf(GREEN "%s" RESET "\n", msg);
36 else if (ft_strcmp(msg, MSG_SLEEP) == 0)
37 printf(BLUE "%s" RESET "\n", msg);
38 else if (ft_strcmp(msg, MSG_THINK) == 0)
39 printf(PURPLE "%s" RESET "\n", msg);
40 else if (ft_strcmp(msg, MSG_DIED) == 0)
41 printf(RED "%s" RESET "\n", msg);
42 else
43 printf("%s\n", msg);
44 }
45 }
46 pthread_mutex_unlock(&philo->table->write_lock);
47}
