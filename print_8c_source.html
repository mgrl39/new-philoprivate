40int write_status(t_philo_status status, t_philo *philo)
41{
42 long time;
43
44 time = gettime(MSEC);
45 time -= philo->table->start_simulation;
46 if (get_int(&philo->philo_mtx, &philo->full))
47 return (0);
48 //safe_mutex_handle(&philo->table->write_mtx, LOCK);
49 if (pthread_mutex_lock(&philo->table->write_mtx))
50 return (ft_alert("ERROR UNLOCK MUTEX", A_ERROR));
51 if (DEBUG_MODE)
52 write_status_debug(status, philo, time);
53 else
54 {
55 if ((TAKE_FIRST_FORK == status || TAKE_SECOND_FORK == status)
56 && !simulation_finished(philo->table))
57 printf(GOLD BOLD"%-6ld"RESET GOLD S_FORK, time, philo->id);
58 else if (EAT == status && !simulation_finished(philo->table))
59 printf(BOLD GREEN "%-6ld"RESET GREEN S_EAT, time, philo->id);
60 else if (SLEEP == status && !simulation_finished(philo->table))
61 printf(BLUE BOLD"%-6ld"RESET BLUE S_SLEEP, time, philo->id);
62 else if (THINK == status && !simulation_finished(philo->table))
63 printf(PURPLE BOLD"%-6ld"RESET PURPLE S_THINK, time, philo->id);
64 else if (DIED == status)
65 printf(BOLD RED"%-6ld" S_DIED RESET, time, philo->id);
66 }
67 // safe_mutex_handle(&philo->table->write_mtx, UNLOCK);
68 // TODO: check if this later...
69 if (pthread_mutex_unlock(&philo->table->write_mtx))
70 return (ft_alert("ERROR UNLOCK MUTEX", A_ERROR));
71 return (0);
72}
