43static void assign_forks(t_philo *philo, t_fork *forks, int philo_position)
44{
45 int philo_nbr;
46
47 philo_nbr = philo->table->philo_nbr;
48 philo->first_fork = &forks[(philo_position + 1) % philo_nbr];
49 philo->second_fork = &forks[philo_position];
50 if (philo->id % 2 == 0)
51 {
52 philo->first_fork = &forks[philo_position];
53 philo->second_fork = &forks[(philo_position + 1) % philo_nbr];
54 }
55}
