60/**
61 * @brief cleans up and frees all allocated resources.
62 *
63 * Destroys all mutexes and frees allocated memory for forks and philosophers.
64 *
65 * @param table Pointer to the shared table structure.
66 */
67void clean_table(t_table *table)
68{
69 int i;
70
71 if (!table)
72 return ;
73 if (table->forks)
74 {
75 i = 0;
76 while (i < table->num_philos)
77 {
78 if (pthread_mutex_destroy(&table->forks[i]) != 0)
79 printf("%s" MSG_FORK_DESTROY_ERR "%s\n", RED, i, RESET);
80 i++;
81 }
82 free(table->forks);
83 }
84 if (pthread_mutex_destroy(&table->write_lock) != 0)
85 printf("%s" MSG_MUTEX_DESTROY_ERR "%s\n", RED, "write lock", RESET);
86 if (pthread_mutex_destroy(&table->meal_lock) != 0)
87 printf("%s" MSG_MUTEX_DESTROY_ERR "%s\n", RED, "meal lock", RESET);
88 if (pthread_mutex_destroy(&table->death_lock) != 0)
89 printf("%s" MSG_MUTEX_DESTROY_ERR "%s\n", RED, "death lock", RESET);
90 if (table->philos)
91 free(table->philos);
92}
