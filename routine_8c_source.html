56/**
57 * Grab the forks: here first and second fork is handy
58 * i dont worry about left right
59 * Eat: write eat, update last meal, update meals counter.
60 * eventyally int full
61 * Release the forks
62 * Set the last meal time:
63 * Actual eating. is set the last meal time and we
64 * are gonna do it immediately.
65 * sleep the time requested.
66 * if is true the philo is full
67 * unlock
68 */
69static int eat(t_philo *philo)
70{
71 if (pthread_mutex_lock(&philo->first_fork->fork))
72 return (ft_alert(F_LOCK_1, A_ERROR));
73 write_status(TAKE_FIRST_FORK, philo);
74 if (pthread_mutex_lock(&philo->second_fork->fork))
75 {
76 if (pthread_mutex_unlock(&philo->first_fork->fork))
77 ft_alert(F_UN_LOCK_2, A_ERROR);
78 return (ft_alert(F_LOCK_2, A_ERROR));
79 }
80 write_status(TAKE_SECOND_FORK, philo);
81 set_long(&philo->philo_mtx, &philo->last_meal_time, gettime(MSEC));
82 philo->meals_counter++;
83 write_status(EAT, philo);
84 precise_usleep(philo->table->time_to_eat, philo->table);
85 if (philo->table->nbr_limit_meals > 0
86 && philo->meals_counter == philo->table->nbr_limit_meals)
87 set_int(&philo->philo_mtx, &philo->full, 1);
88 if (pthread_mutex_unlock(&philo->first_fork->fork))
89 return (ft_alert(F_UNLOCK_1, A_ERROR));
90 if (pthread_mutex_unlock(&philo->second_fork->fork))
91 return (ft_alert(F_UNLOCK_2, A_ERROR));
92 return (0);
93}

108void *dinner_simulation(void *data)

130 }

134/**
135 * If no meals, return
136 * If only one philo -> function for this situation
137 * Create all threads (all philos)
138 * Create a monitor thread -> that is searching for philos which are death.
139 * Syncronize the beggining of the simulation -> everytime you call
140 * pthread_create the philosophers start to run.
141 * I want every philo start simultaneously.
142 * We need synchronization
143 * thing to make all the philos start at the same time.
144 * join (wait) everyone.
145 */
146int dinner_start(t_table *table)
147{
148 int created;
149
150 if (0 == table->nbr_limit_meals)
151 return (SUCCESS);
152 if (create_philos(table, &created))
153 {
154 set_int(&table->table_mtx, &table->end_simulation, 1);
155 set_int(&table->table_mtx, &table->all_threads_ready, 1);
156 return (join_philos(table, created), FAILURE);
157 }
158 if (pthread_create(&table->monitor, NULL, monitor_dinner, table))
159 {
160 ft_alert(F_CREAT_MONITOR_TH, A_ERROR);
161 set_int(&table->table_mtx, &table->end_simulation, 1);
162 set_int(&table->table_mtx, &table->all_threads_ready, 1);
163 return (join_philos(table, created), FAILURE);
164 }
165 table->start_simulation = gettime(MSEC);
166 set_int(&table->table_mtx, &table->all_threads_ready, 1);
167 join_philos(table, table->philo_nbr);
168 set_int(&table->table_mtx, &table->end_simulation, 1);
169 if (pthread_join(table->monitor, NULL))
170 ft_alert(F_JOIN_MONITOR_THR, A_WARN);
171 return (SUCCESS);
172}
