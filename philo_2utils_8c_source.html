17/*
18 * We are gonna exploit gettimeofday
19 *
20 * time_code -> SECONDS MILLISECONDS MICROSECONDS
21 *
22 * Is gonna set the seconds and the microseconds
23 */
24long gettime(t_time_code time_code)
25{
26 struct timeval tv;
27
28 if (gettimeofday(&tv, NULL))
29 error_exit("Gettimeofday failed!");
30 if (SECOND == time_code)
31 return (tv.tv_sec + (tv.tv_usec / 1e6));
32 else if (MILLISECOND == time_code)
33 return ((tv.tv_sec * 1e3) + tv.tv_usec / 1e3);
34 else if (MICROSECOND == time_code)
35 return ((tv.tv_sec * 1e6) + tv.tv_usec);
36 else
37 error_exit("Wrong input to gettime!");
38 return (1337);
39}

41/**
42 * precise usleep, the real one suck.

49 * 1) USLEEP THE MAJORITY OF TIME, NOT CPU INTENSIVE

52 * This will give us more precise than the actual system function usleep

54 */

56void precise_usleep(long usec, t_table *table)
57{
58 long start;
59 long elapsed;
60 long remaining;
61
62 start = gettime(MICROSECOND);
63 while (gettime(MICROSECOND) - start < usec)
64 {
65 if (simulation_finished(table))
66 break ;
67 elapsed = gettime(MICROSECOND) - start;
68 remaining = usec - elapsed;
69 if (remaining > 1e3)
70 usleep(remaining / 2);
71 else
72 {
73 while (gettime(MICROSECOND) - start < usec)
74 ;
75 }
76 }
77}
