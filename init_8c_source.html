56static void assign_forks(t_philo *philo, t_fork *forks, int philo_position)
57{
58 int philo_nbr;
59
60 philo_nbr = philo->table->philo_nbr;
61 philo->first_fork = &forks[(philo_position + 1) % philo_nbr];
62 philo->second_fork = &forks[philo_position];
63 if (philo->id % 2 == 0)
64 {
65 philo->first_fork = &forks[philo_position];
66 philo->second_fork = &forks[(philo_position + 1) % philo_nbr];

130int init_table(t_table *table)
131{
132 int i;
133
134 table->philos = (t_philo *)malloc(sizeof(t_philo) * table->philo_nbr);
135 if (!table->philos)
136 return (ft_alert(ERR_MALLOC, A_ERROR));
137 if (pthread_mutex_init(&table->table_mutex, NULL) != 0)
138 return (free_table(table, 0), ft_alert(ERR_MUTEX, A_ERROR));
139 if (pthread_mutex_init(&table->write_mutex, NULL) != 0)
140 return (free_table(table, 0), ft_alert(ERR_MUTEX, A_ERROR));
141 table->forks = (t_fork *)malloc(sizeof(t_fork) * table->philo_nbr);
142 if (!table->forks)
143 return (free_table(table, 0), ft_alert(ERR_MALLOC, A_ERROR));
144 i = -1;
145 while (++i < table->philo_nbr)
146 {
147 if (pthread_mutex_init(&table->forks[i].fork, NULL) != 0)
148 return (free_table(table, i), ft_alert(ERR_MUTEX, A_ERROR));
149 table->forks[i].fork_id = i;
150 }
151 if (philo_init(table) == FAILURE)
152 return (free_table(table, table->philo_nbr), FAILURE);
153 return (SUCCESS);
154}
