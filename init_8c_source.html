44static void assign_forks(t_philo *philo, t_fork *forks, int philo_position)
45{
46 int philo_nbr;
47
48 philo_nbr = philo->table->philo_nbr;
49 philo->first_fork = &forks[(philo_position + 1) % philo_nbr];
50 philo->second_fork = &forks[philo_position];
51 if (philo->id % 2 == 0)
52 {
53 philo->first_fork = &forks[philo_position];
54 philo->second_fork = &forks[(philo_position + 1) % philo_nbr];
55 }
56}
