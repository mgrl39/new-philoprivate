15/**
16 * Initializes the mutexes required for the simulation
17 *
18 * Allocates and initializes one mutex per fork, as well as separate mutexes for
19 * writing output, meal tracking, and death detection.
20 *
21 * @param table Pointer to the shared table structure
22 * @return 0 on succes, 1 on failure
23 */
24int init_mutexes(t_table *table)
25{
26 int i;
27
28 debug_print("Reservando memoria para tenedores");
29 table->forks = malloc(sizeof(pthread_mutex_t) * table->num_philos);
30 if (!table->forks)
31 {
32 debug_print("Error en malloc de tenedores");
33 return (ft_error(MSG_MALLOC_ERR));
34 }
35 i = 0;
36 while (i < table->num_philos)
37 {
38 debug_print("Inicializando mutex para tenedor %d", i);
39 if (pthread_mutex_init(&table->forks[i], NULL))
40 {
41 debug_print("Error al inicializar mutex de tenedor %d", i);
42 return (ft_error(MSG_MALLOC_ERR));
43 }
44 i++;
45 }
46 debug_print("inicializando mutex de escritura");
47 if (pthread_mutex_init(&table->write_lock, NULL))
48 {
49 debug_print("Error al inicializar mutex de escritura");
50 return (ft_error(MSG_MALLOC_ERR));
51 }
52 if (pthread_mutex_init(&table->meal_lock, NULL))
53 {
54 debug_print("Error al inciailziar mutex de comidas");
55 return (ft_error(MSG_MALLOC_ERR));
56 }
57 if (pthread_mutex_init(&table->death_lock, NULL))
58 {
59 debug_print("Error al inicailizar mutex de muerte");
60 return (ft_error(MSG_MALLOC_ERR));
61 }
62 debug_print("Todos los mutexes inicializados correctamente");
63 return (0);
64}

66/**
67 * Initializes the philosoher structures/
68 *
69 * Allocates memory for each philosopher and sets their IDs,
70 * pointer to the shared table
71 * and assigns the correspondign left and right forks (mutex pointers).
72 *
73 * @param table Pointer to the shared table structure.
74 * @return 0 on succes, 1 on failure
75 */
76int init_philos(t_table *table)
77{
78 int i;
79
80 debug_print("Reservado memoria para filosofos");
81 table->philos = malloc(sizeof(t_philo) * table->num_philos);
82 if (!table->philos)
83 {
84 debug_print("Error en malloc de filosofos");
85 return (ft_error(MSG_MALLOC_ERR));
86 }
87 i = 0;
88 debug_print("Inicializando memoria de filsofs con ceros");
89 memset(table->philos, 0, sizeof(t_philo) * table->num_philos);
90 while (i < table->num_philos)
91 {
92 debug_print("Configurando filosofo %d", i + 1);
93 table->philos[i].id = i + 1;
94 table->philos[i].table = table;
95 table->philos[i].meals_eaten = 0;
96 table->philos[i].last_meal_time = table->start_time;
97 table->philos[i].forks[LEFT] = &table->forks[i];
98 table->philos[i].forks[RIGHT] = &table->forks[(i + 1) % table->num_philos];
99 if (table->num_philos == 1)
100 {
101 debug_print("Solo hay un filosofo, fijando tenedor derecho a NULL");
102 table->philos[i].forks[RIGHT] = NULL;
103 }
104 i++;
105 }
106 return (0);
107}
