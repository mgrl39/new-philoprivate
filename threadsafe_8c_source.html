17/*
18 * Embed controls on return status
19 * pthread_mutex_init pthread_mutex_unlock pthread_mutex_lock
20 * and pthread_mutex_init()
21 * This functions returns 0 if successful, otherwise we will get an error value.
22 * We will recreate the perror function
23 */
24static void handle_mutex_error(int status, t_opcode opcode)
25{
26 if (0 == status)
27 return ;
28 else if (EINVAL == status && (LOCK == opcode || UNLOCK == opcode))
29 error_exit("The value specified by mutex is invalid.");
30 else if (EDEADLK == status)
31 error_exit(
32 "A deadlock would occur if the thread blocked waiting for mutex.");
33 else if (EPERM == status)
34 error_exit("The current thread does not hold a lock on muttex.");
35 else if (ENOMEM == status)
36 error_exit(
37 "The process cannot allocate enogh memory to create antoher mutex."
38 );
39 else if (EBUSY == status)
40 error_exit("Mutex is locked");
41}

49void safe_mutex_handle(t_mtx *mutex, t_opcode opcode)

64{

69 else if (EPERM == status)
70 error_exit("The caller does not have appropiate permission\n");

72 error_exit("The value specified by attr is invalid.");
73 else if (EINVAL == status && JOIN == opcode)
74 error_exit("The value specified by thread is not joinable\n");
75 else if (ESRCH == status)
76 error_exit("No thread could be fould corresponding to that \

78 else if (EDEADLK == status)
