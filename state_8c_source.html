16 * Module containing functions to avoid writing LOCK UNLOCK everywhere
17 * These functions now return int for error handling
18 * Return SUCCESS (0), FAILURE (1) for errors.
19 */
20int set_int(t_mtx *mutex, int *dest, int value)
21{
22 if (pthread_mutex_lock(mutex))
23 return (ft_alert(FAIL_LOCK_SET_INT, A_ERROR));
24 *dest = value;
25 if (pthread_mutex_unlock(mutex))
26 return (ft_alert(FAIL_UNLOCK_SET_INT, A_ERROR));

28}
