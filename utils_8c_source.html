90void print_status(t_philo *philo, char *msg)
91{
92 long timestamp;
93
94 if (!philo || !msg || !philo->table)
95 return ;
96 pthread_mutex_lock(&philo->table->write_lock);
97 if (!check_death_flag(philo->table))
98 {
99 timestamp = get_time();
100 if (timestamp != -1)
101 {
102 timestamp -= philo->table->start_time;
103 printf("%06ld %d ", timestamp, philo->id);
104 if (ft_strcmp(msg, MSG_FORK) == 0)
105 printf(GOLD "%s" RESET "\n", msg);
106 else if (ft_strcmp(msg, MSG_EAT) == 0)
107 printf(GREEN "%s" RESET "\n", msg);
108 else if (ft_strcmp(msg, MSG_SLEEP) == 0)
109 printf("\033[38;5;75m%s" RESET "\n", msg);
110 else if (ft_strcmp(msg, MSG_THINK) == 0)
111 printf("\033[38;5;147m%s" RESET "\n", msg);
112 else if (ft_strcmp(msg, MSG_DIED) == 0)
113 printf(RED "%s" RESET "\n", msg);
114 else
115 printf("%s\n", msg);
116 }
117 }
118 pthread_mutex_unlock(&philo->table->write_lock);
119}
